describe('ClaimWorkflow E2E', () => {

        .post('/api/v1/claims')

      }, { timeout: 1000 });
    });

      const response = await request(DICT_API_URL)
        .post(`/api/v1/claims/${claimId}/confirm`)
        .set('Authorization', `Bearer ${ownerToken}`);

      expect(response.body.status).toBe('CONFIRMED');
